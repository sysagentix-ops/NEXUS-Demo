package PortableSolarLabStation_Modular {
    
    private import ScalarValues::*;
    private import SI::*;
    private import ISQMechanics::*;
    private import ISQElectromagnetism::*;
    private import ISQ::*;
    private import ISQThermodynamics::*;

    // --- 1. Type Aliases ---
    attribute def LocalMass              :> MassValue;
    attribute def LocalElectricPotential :> ElectricPotentialValue;
    attribute def LocalElectricCurrent    :> ElectricCurrentValue;
    attribute def LocalPower              :> PowerValue;
    attribute def LocalTemperature        :> TemperatureValue;
    attribute def LocalElectricCharge     :> ElectricChargeValue;

    // --- 2. Item Definitions (The "Medium") ---
    item def Electricity;
    item def ModbusPacket;

    // --- 3. Base Port Definitions ---
    // UNIDIRECTIONAL: Uses 'out' - can only send
    port def PowerPort {
        out item powerOut : Electricity;
    }

    port def DataPort {
        out item dataOut : ModbusPacket;
    }

    // BIDIRECTIONAL: Uses 'inout' - can send AND receive
    port def BidirectionalPowerPort {
        inout item powerIO : Electricity;  // ← 'inout' keyword for bidirectional!
    }

    port def BidirectionalDataPort {
        inout item dataIO : ModbusPacket;  // ← 'inout' keyword for bidirectional!
    }

    // --- 4. Specialized Interface Definitions ---
    // Unidirectional: source sends, sink receives
    interface def PowerFlowInterface {
        end port source : PowerPort;
        end port sink : ~PowerPort;
    }

    // Bidirectional: both ends can send and receive
    interface def BidirectionalPowerInterface {
        end port portA : BidirectionalPowerPort;
        end port portB : ~BidirectionalPowerPort;
    }

    interface def BidirectionalDataInterface {
        end port master : BidirectionalDataPort;
        end port slave : ~BidirectionalDataPort;
    }

    // --- 5. Physical Connection Definitions ---
    // Unidirectional power link (solar → battery)
    connection def AndersonPP45UnidirectionalLink specializes PowerFlowInterface {
        attribute connectorType = "Anderson PP45";
        attribute maxCurrent : LocalElectricCurrent = 45 [A];
        attribute gauge : Real = 12; // AWG
    }

    // Bidirectional power link (battery ↔ regulator: battery charges and discharges)
    connection def AndersonPP45BidirectionalLink specializes BidirectionalPowerInterface {
        attribute connectorType = "Anderson PP45";
        attribute maxCurrent : LocalElectricCurrent = 45 [A];
        attribute gauge : Real = 12; // AWG
    }

    // Bidirectional data link (Modbus RTU: send and receive commands/data)
    connection def ModbusRTULink specializes BidirectionalDataInterface {
        attribute medium = "Shielded Twisted Pair";
        attribute baudRate : Real = 9600;
        attribute protocol = "Modbus RTU";
    }

    // --- 6. Component Definitions ---

    part def SolarInputModule {
        attribute electrical_voltage : LocalElectricPotential = 19.8 [V];
        attribute electrical_powerRated : LocalPower = 300 [W];
        attribute thermal_operatingTemp : LocalTemperature;

        assert constraint { thermal_operatingTemp >= 253.15 [K] and thermal_operatingTemp <= 338.15 [K] }

        port pwrOut : PowerPort;  // Unidirectional output (only sends power)
        port dataPort : BidirectionalDataPort;  // Bidirectional (sends status, receives commands)
    }

    part def LiFePO4BatteryBank {
        attribute physical_mass : LocalMass = 24 [kg];
        attribute electrical_voltage : LocalElectricPotential = 12.8 [V];
        attribute electrical_capacity : LocalElectricCharge = 720000 [C]; // 200Ah * 3600

        port pwrIn : ~PowerPort;  // Unidirectional input (receives charge from solar)
        port pwrOut : BidirectionalPowerPort;  // Bidirectional (charges from regulator, discharges to regulator)
        port dataPort : BidirectionalDataPort;  // Bidirectional (sends status, receives commands)
    }

    part def DCOutputRegulator {
        attribute electrical_voltage : LocalElectricPotential = 12 [V];
        attribute electrical_currentMax : LocalElectricCurrent = 20 [A];
        attribute performance_efficiency : Real = 0.95;

        port batteryPort : BidirectionalPowerPort;  // Bidirectional (charges battery, receives from battery)
    }

    part def SystemMonitoringUnit {
        attribute data_protocol : String = "Modbus RTU";

        // Bidirectional data ports (sends commands, receives status/data)
        port solarMon : BidirectionalDataPort;
        port batteryMon : BidirectionalDataPort;
    }

    // --- 7. System Assembly (The Station) ---

    part def PortableSolarLabStationPrototype {
        part solar : SolarInputModule;
        part battery : LiFePO4BatteryBank;
        part regulator : DCOutputRegulator;
        part monitor : SystemMonitoringUnit;

        // UNIDIRECTIONAL Power Connection (solar sends to battery, one-way flow)
        connection : AndersonPP45UnidirectionalLink 
            connect solar.pwrOut to battery.pwrIn;

        // BIDIRECTIONAL Power Connection (battery ↔ regulator: charges and discharges)
        connection : AndersonPP45BidirectionalLink 
            connect battery.pwrOut to regulator.batteryPort;

        // BIDIRECTIONAL Data Connections (Modbus RTU: commands and status exchange)
        connection : ModbusRTULink 
            connect monitor.solarMon to solar.dataPort;

        connection : ModbusRTULink 
            connect monitor.batteryMon to battery.dataPort;
    }

    part stationInstance : PortableSolarLabStationPrototype;
}
