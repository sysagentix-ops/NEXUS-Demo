package PortableSolarLabStation_Modular {
    
    private import ScalarValues::*;
    private import SI::*;
    private import ISQMechanics::*;
    private import ISQElectromagnetism::*;
    private import ISQ::*;
    private import ISQThermodynamics::*;

    // --- 1. Type Aliases ---
    attribute def LocalMass              :> MassValue;
    attribute def LocalElectricPotential :> ElectricPotentialValue;
    attribute def LocalElectricCurrent    :> ElectricCurrentValue;
    attribute def LocalPower              :> PowerValue;
    attribute def LocalTemperature        :> TemperatureValue;
    attribute def LocalElectricCharge     :> ElectricChargeValue;

    // --- 2. Item Definitions (The "Medium") ---
    item def Electricity;
    item def ModbusPacket;

    // --- 3. Base Port Definitions ---
    port def PowerPort {
        out item electricity : Electricity;
    }

    port def DataPort {
        out item packet : ModbusPacket;
    }

    // --- 4. Specialized Interface Definitions ---
    // Logical: Defines the direction and type of flow
    interface def PowerFlowInterface {
        end port source : PowerPort;
        end port sink : ~PowerPort;
    }

    interface def DataFlowInterface {
        end port master : DataPort;
        end port slave : ~DataPort;
    }

    // --- 5. Physical Connection Definitions ---
    // Specialization: Adds physical constraints to the interface
    connection def AndersonPP45Link specializes PowerFlowInterface {
        attribute connectorType = "Anderson PP45";
        attribute maxCurrent : LocalElectricCurrent = 45 [A];
        attribute gauge : Real = 12; // AWG
    }

    connection def ModbusRTULink specializes DataFlowInterface {
        attribute medium = "Shielded Twisted Pair";
        attribute baudRate : Real = 9600;
        attribute protocol = "Modbus RTU";
    }

    // --- 6. Component Definitions ---

    part def SolarInputModule {
        attribute electrical_voltage : LocalElectricPotential = 19.8 [V];
        attribute electrical_powerRated : LocalPower = 300 [W];
        attribute thermal_operatingTemp : LocalTemperature;

        assert constraint { thermal_operatingTemp >= 253.15 [K] and thermal_operatingTemp <= 338.15 [K] }

        port pwrOut : PowerPort;
        port dataOut : DataPort;
    }

    part def LiFePO4BatteryBank {
        attribute physical_mass : LocalMass = 24 [kg];
        attribute electrical_voltage : LocalElectricPotential = 12.8 [V];
        attribute electrical_capacity : LocalElectricCharge = 720000 [C]; // 200Ah * 3600

        port pwrIO : ~PowerPort; // It receives charge from the panel
        port dataOut : DataPort;
    }

    part def DCOutputRegulator {
        attribute electrical_voltage : LocalElectricPotential = 12 [V];
        attribute electrical_currentMax : LocalElectricCurrent = 20 [A];
        attribute performance_efficiency : Real = 0.95;

        port pwrIn : ~PowerPort;
        port pwrOut : PowerPort;
        port dataOut : DataPort;
    }

    part def SystemMonitoringUnit {
        attribute data_protocol : String = "Modbus RTU";

        // Multiple "Slave" ports to receive data from components
        port solarMon : ~DataPort;
        port batteryMon : ~DataPort;
        port regulatorMon : ~DataPort;
    }

    // --- 7. System Assembly (The Station) ---

    part def PortableSolarLabStationPrototype {
        part solar : SolarInputModule;
        part battery : LiFePO4BatteryBank;
        part regulator : DCOutputRegulator;
        part monitor : SystemMonitoringUnit;

        // Specialized Power Connections
        connection : AndersonPP45Link 
            connect solar.pwrOut to battery.pwrIO;

        connection : AndersonPP45Link 
            connect battery.pwrIO to regulator.pwrIn;

        // Specialized Data Connections
        connection : ModbusRTULink 
            connect monitor.solarMon to solar.dataOut;

        connection : ModbusRTULink 
            connect monitor.batteryMon to battery.dataOut;

        connection : ModbusRTULink 
            connect monitor.regulatorMon to regulator.dataOut;
    }

    part stationInstance : PortableSolarLabStationPrototype;
}
