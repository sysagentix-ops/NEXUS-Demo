STRICT TYPING & UNIT RULES

    The Quantity Rule: If an attribute has a unit literal (e.g., = 10 [kg]), it MUST be typed with a local alias (e.g., : LocalMass).

    The Real Rule: Use : Real ONLY for dimensionless numbers, ratios, or counts where no unit brackets [] are used.

    The ASCII Rule: Use only ASCII characters. Never use Ω. Use Ohm.

    The SI Library Rule: Use only standard SI units. bar is NOT supported. Convert bar to Pa (1 bar=100000 [Pa]).

    The Percent Rule: Do not use %. Convert to a decimal Real (e.g., 0.95 instead of 95%).

    private import ScalarValues::*;

    private import ISQMechanics::*;
    private import ISQElectromagnetism::*;
    private import ISQ::*;
    private import ISQThermodynamics::*;

    // Only the types actually used
    attribute def LocalMass              :> MassValue;
    attribute def LocalElectricPotential :> ElectricPotentialValue;
    attribute def LocalElectricCurrent   :> ElectricCurrentValue;
    attribute def LocalPower             :> PowerValue;
    attribute def LocalTemperature       :> TemperatureValue;

    part def SolarInputModule {
        attribute name = "SolarInputModule";
        attribute type = "active";
        attribute description = "Portable 300 W solar input with MPPT tracking";
        attribute count = 1;
        attribute parentComponent = null;

        attribute electrical_voltage : LocalElectricPotential = 19.8 [V];
        attribute electrical_currentMax : LocalElectricCurrent = 18 [A];
        attribute electrical_powerRated : LocalPower = 300 [W];

        attribute thermal_operatingTemp : LocalTemperature;

        assert constraint {
            thermal_operatingTemp >= 253.15 [K] and
            thermal_operatingTemp <= 338.15 [K]
        } doc /* Solar module temperature range */

        attribute subcomponents = "";
    }

    part def LiFePO4BatteryBank {
